function * range (limit) {
  for (let i = 0; i < limit; i++) {
    yield i
  }
}

class List {
  constructor(first, rest) {
    this.first = first
    this.rest = rest
  }

  *[Symbol.iterator]() {
    let node = this
    while(node) {
      yield node.first
      node = node.rest
    }
  }
}
